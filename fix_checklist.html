def __init__(self, constraint_matrix=None, **kwargs):
    super().__init__(**kwargs)
    self.constraint_matrix = constraint_matrix  # Tensor (num_acronyms, num_classes) bool

def compute_loss(self, model, inputs, return_outputs=False, **kwargs):
    labels = inputs.pop("labels")
    acronym_ids = inputs.pop("acronym_ids", None)
    outputs = model(**inputs)
    logits = outputs.logits

    # Apply constraint mask trước khi tính loss
    if acronym_ids is not None and self.constraint_matrix is not None:
        mask = self.constraint_matrix[acronym_ids].to(logits.device)  # (B, C)
        logits = logits.masked_fill(~mask, float("-inf"))

    loss = self.focal_loss_fn(logits, labels)
    return (loss, outputs) if return_outputs else loss

def compute_metrics_detailed(eval_pred, eval_dataset, id2expansion, expansion2id):
    logits, labels = eval_pred
    preds = np.argmax(logits, axis=-1)

    seen_mask, unseen_mask = [], []
    for label in labels:
        exp = id2expansion.get(int(label), "")
        seen = exp in expansion2id  # có trong train không?
        seen_mask.append(seen)
        unseen_mask.append(not seen)

    seen_mask = np.array(seen_mask)
    return {
        "acc_overall": (preds == labels).mean(),
        "acc_seen":    (preds[seen_mask] == labels[seen_mask]).mean(),
        "acc_unseen":  (preds[unseen_mask] == labels[unseen_mask]).mean()
            if unseen_mask.any() else 0.0,
    }

Medium #4 train_acronym.py
Focal Loss + Label Smoothing tương tác không rõ ràng
Với 280 classes, smooth mass per class ≈ 0.00036 — quá nhỏ để có nghĩa. Gamma của focal loss suy yếu trên soft targets
⚡ Impact
label_smoothing=0.1 phân phối 10% mass ra 280 classes → mỗi "wrong" class chỉ nhận 0.036%. Focal weight (1−p)^2 tính dựa trên soft targets sẽ cho giá trị sai so với hard targets. Hai kỹ thuật tranh nhau thay vì cộng hưởng.

# Adaptive threshold dựa trên số valid choices
def _get_threshold(self, acronym: str, default: float = 0.35) -> float:
    valid_ids = self.acronym_constraint.get(acronym.lower(), [])
    n = len(valid_ids)
    if n <= 2: return 0.55  # 2 choices → expect confident
    if n <= 4: return 0.40
    return 0.30               # 5-7 choices → lower bar
